function reverse(originalStr) {
  var strArray = originalStr.split("");
  var reverseArray = [];
  for (var i=strArray.length-1; i>=0; i--) {
    reverseArray.push(strArray[i]);
  }
  return reverseArray.join("");
}

class Queue {
  constructor () {
    arr = new Array();
  }

  function enqueue (element) {
    this.arr.push(element);
  }

  function dequeue {
    let temp = this.arr[0];
    this.arr.shift();
    return temp;
  }
}
